num_simulations <- 10


play_game <- function() {
  dice_result <- sample(1:6, size = 1)
  if (dice_result %% 2 == 0) {
    return(10)
  } else {
    return(15)
  }
}


game_results <- replicate(num_simulations, play_game())


average_amount <- mean(game_results)

Maximum age is when he can celebrate his birthdat at 12 oclock in the morning: which is 51

SurvivalRate of year n (denote as SVraten) = 1 - (n-20)* 2%

Simple life insurance Consider a 20-year-old male born in 01/01/2004. Suppose the survival rate is reduced 2% per year. That means survival rate at age 21 is 98%, and that at age 22 is 96%, etc. In other words, the probability the person dies within age 21 is 2%, within age 22 is 4%, etc. The person will die when the survival rate is at 0%. A simple life insurance product will pay him 10000 USD if he dies within age 40. Assume that the risk free interest rate is 7% compounded continuously. a) What is the probability that he dies within at age 40?

A. Prop_dead_40 = prop_survive_39 dead_rate_40 = product of (SVrate20 Svrate21 SVrate22 ...SVrate59) (40-20)*2% = 0.004809220638

B. expected Amount received = Prop_dead_40 payment = 0.004809220638 10000 = 48.09 (USD)

c) What is the fair price for this product? Fair Price = expected value discount factor = 48.09 EXP(-8%*(40-20)) = 6.508572372 USD

# Initialize vectors to store data
age <- start_age:end_age
year_from_age <- age - start_age
survival_rate <- 1 - (year_from_age * annual_reduction_rate)
prop_of_survival <- cumprod(survival_rate)
dead_rate <- year_from_age * annual_reduction_rate
prop_of_dead <- c(0, prop_of_survival[-length(prop_of_survival)] * dead_rate[-1])


expected_payoff <- payout * prop_of_dead
present_value <- expected_payoff / exp(discount_rate * year_from_age)


life_table <- data.frame(
  age = age,
  year_from_age = year_from_age,
  survival_rate = survival_rate,
  prop_of_survival = prop_of_survival,
  dead_rate = dead_rate,
  prop_of_dead = prop_of_dead,
  expected_payoff = expected_payoff,
  present_value = present_value
